\frametitle{Перевод из СС с основанием 2 в СС с основанием 4}

\textbf{Сложный путь: } 1) CC-2 -> CC-10: $10100_{(2)} = 20_{(10)}$ \\
\qquad\qquad\qquad\qquad 2) CC-10 -> CC-4: $20_{(10)} = 110_{(4)} => 10100_{(2)} = 110_{4)}$\\

Примечание: ''СС-N'' означает ''система счисления с основанием N''\\
\vspace{3em}
\textbf{Простой путь:}

\begin{center}
	$x_{i+1}2^{i+1} + x_{i}2^{i} + \cdots + x_{3}2^{3} + x_{2}2^{2} + x_{1}2^{1} + x_{0}2^{0}$\\
	
	$\Downarrow$	
	
	$x_{2k+1}2^{2k+1} + x_{2k}2^{2k} + \cdots + x_{3}2^{2 \cdot 1 + 1} + x_{2}2^{2 \cdot 1} + x_{1}2^{1} + x_{0}2^{0}$\\
	
	$\Downarrow$	
	
	$2^{2k}(x_{2k+1}2^{1} + x_{2k}) + \cdots + 2^{2}(x_{3}2^{1} + x_{2}) + 2^0(x_{1}2^1 + x_{0})$
	
	$\Downarrow$
	
	$4^k(x_{2k+1}2^{1} + x_{2k}) + \cdots + 4^{1}(x_{3}2^{1} + x_{2}) + 4^{0}(x_{1}2^{1} + x_{0})$
	
\end{center}

\end{frame}

\begin{frame}
	
	\frametitle{Преобразование из CC-N в CC-$N^{k}$ и обратно}
	\footnotesize
	\textcolor{coolgreen}{Из СС-N в СС-$N^{k}$}
	\begin{itemize}
		\item дополнить число, записанное в СС с основанием N, незначащими нулями так, чтобы количество цифр было кратно k;
		\item разбить полученное число на группы по k цифр, начиная от нуля;
		\item заменить каждую такую группу эквивалентным числом, записанным в СС с основанием N
	\end{itemize}
	\qquad\qquad\qquad\qquad Задача: $1020101_{(3)} = ?_{(27)}$\\
	\qquad\qquad\qquad\qquad Решение: $1020101_{(3)} = 001 020 101_{(3)} = 16A?_{(27)}$\\
	
	\textcolor{coolgreen}{Из СС-$N^{k}$ в СС-N}
	\begin{itemize}
		\item заменить каждую цифру числа, записанного в CC с основанием N$^{k}$, эквивалентным набором из k цифр CC с основанием N.
	\end{itemize}
	\qquad\qquad\qquad\qquad Задача: $1020101_{(3)} = ?_{(27)}$\\
	\qquad\qquad\qquad\qquad Решение: $1020101_{(3)} = 001 020 101_{(3)} = 16A?_{(27)}$
