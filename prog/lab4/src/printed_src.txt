     1	package application.extensions;
     2	
     3	import engine.interfaces.Renderable;
     4	import engine.scene.nodes.DrawableNode;
     5	import lombok.*;
     6	import engine.structs.IntVector2;
     7	
     8	@Data
     9	@AllArgsConstructor
    10	@NoArgsConstructor
    11	public class SpriteNode extends DrawableNode<Screen> implements Renderable<Screen> {
    12	
    13	    SpriteData sprite;
    14	
    15	    public void renderToScreen(Screen screen) {
    16	        if (!isVisible()|| sprite == null) return;
    17	
    18	        IntVector2 resolution = sprite.getResolution();
    19	        IntVector2 screenBoundaries = screen.getResolution();
    20	
    21	        IntVector2 globalPosition = getGlobalPosition();
    22	
    23	        char[][] screenCanvas = screen.getCanvas();
    24	
    25	        for (int x = 0; x < resolution.X; x++) {
    26	            for (int y = 0; y < resolution.Y; y++) {
    27	                char currentChar = sprite.getCurrentSprite()[y][x];
    28	
    29	                if (currentChar == ' ') {
    30	                    continue;
    31	                }
    32	                if (globalPosition.Y + y < 0 ||
    33	                        globalPosition.X + x < 0 ||
    34	                        globalPosition.Y + y >= screenBoundaries.Y ||
    35	                        globalPosition.X + x >= screenBoundaries.X) {
    36	                    continue;
    37	                }
    38	
    39	                screenCanvas[globalPosition.Y + y][globalPosition.X + x] = currentChar;
    40	            }
    41	        }
    42	    }
    43	}
